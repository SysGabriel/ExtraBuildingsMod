event = {
	
	id = EBM_tech_centre.0
	title = EBM_tech_centre.0.t
	desc = EBM_tech_centre.0.d
	
	trigger = {
		has_built = {
			type = tech_centre
			value > 1
		}
		NOT = {
			has_idea = EBM_tech_centre_idea_built
		}
		
	}

	option = {
		name = EBM_tech_centre.0.a
		every_country = {
			if = {
				limit = {
					NOT = {
						has_idea = EBM_tech_centre_idea_built
						has_built = {
							type = tech_centre
							value > 1
						}
					}					
				}
				add_ideas = EBM_tech_centre_idea_behind
			}
		}	
	}

	option = {
		name = EBM_tech_centre.0.b
		every_country = {
			if = {
				limit = {
					has_idea = EBM_tech_centre_idea_built
				}				
				add_stability = 0.01
			}
		}
	}

	option = {
		name = EBM_tech_centre.0.c
		if = {
			limit = {
				has_built = {
					type = tech_centre
					value > 1
				}
				NOT = {
					has_idea = EBM_tech_centre_idea_built
				}
			}
			add_ideas = EBM_tech_centre_idea_built			
		}
		
	}
}
