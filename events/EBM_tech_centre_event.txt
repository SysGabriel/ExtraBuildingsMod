# First Event
country_event = {
	
	id = EBMTC.1
	title = EBMTC.1.t
	desc = EBMTC.1.d

	fire_only_once = yes
	mean_time_to_happen = {
		months = 1
	}

	trigger = {
		has_built = {
			type = tech_centre
			value = 1
		}
		NOT = {
			has_idea = EBM_tech_centre_idea_built
		}
		
	}
 
	option = {
		name = EBMTC.1.a
		add_ideas = EBM_tech_centre_idea_built
		remove_ideas = EBM_tech_centre_idea_behind
		set_variable = {
			var = has_tech_centre
			value = 1 #Adding a global variable for the next events 1 to true and 0 to false
		}
		hidden_effect = {
			every_country = {
				if = {
					limit = {
						NOT = {
							has_idea = EBM_tech_centre_idea_built
						}
					}
					news_event = EBMTC.2
				}
				else = {
					news_event = EBMTC.3
				}
			}
		}
	}	
}

news_event = {
	id = EBMTC.2
	title = EBMTC.2.t
	desc = EBMTC.2.d

	is_triggered_only = yes

	option = {
		name = EBMTC.2.a
		hidden_effect = {
			add_ideas = EBM_tech_centre_idea_behind
		}
	}
	
}

news_event = {
	id = EBMTC.3
	title = EBMTC.2.t
	desc = EBMTC.2.d

	is_triggered_only = yes

	option = {
		name = EBMTC.3.a
		hidden_effect = {
			add_stability = 0.01
			add_war_support = -0.1
		}
	}
	
}

# Second Event - Focus Selection
country_event = {
	
	id = EBMTC.4
	title = EBMTC.4.t
	desc = EBMTC.4.d
	fire_only_once = yes
	
	mean_time_to_happen = {
		months = 10
	}

	trigger = {
		check_variable = {
			var = has_tech_centre
			value = 1
			compare = equals
		}
	}
	
	option = {
		name = EBMTC.4.a
		hidden_effect = {
			set_variable = {
				var = tech_centre_focus
				value = "civilian"
			}
			add_ideas = EBM_tech_centre_idea_civ_focus
		}
	}

	option = {
		name = EBMTC.4.b
		hidden_effect = {
			set_variable = {
				var = tech_centre_focus
				value = "military"
			}
			add_ideas = EBM_tech_centre_idea_mil_focus
		}
	}

	option = {
		name = EBMTC.4.c
		hidden_effect = {
			set_variable = {
				var = tech_centre_focus
				value = "espionage"
			}
			add_ideas = EBM_tech_centre_idea_spy_focus
		}
	}


}
